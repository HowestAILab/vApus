% Super- and subscript
\newcommand{\supsc}[1]{\ensuremath{^{\text{#1}}}}
\newcommand{\subsc}[1]{\ensuremath{_{\text{#1}}}}

% Command to add images. Usage:
% \img[H]{width=5cm}{filename}{caption}
\newcommand{\img}[4][!ht]{                  % The first argument is default 'ht'.
    \begin{figure}[#1]
		\begin{center}
            \includegraphics[#2]{#3}
            \caption{#4\label{#3}}
        \end{center}
    \end{figure}
}

% New command to add tables. Usage:
% \newtable[H]{filename-tabular}{caption}    
\newcommand{\newtable}[3][h]{
    \begin{table}[#1]                      % The file must contain one element: \begin{table}...\end{table}.
        \begin{center}
            \input{tables/#2}
			\caption{#3\label{#2}}
        \end{center}
    \end{table}
}

% New command to add small tables. Usage:
% \smalltable[H]{filename-tabular}{caption}    
\newcommand{\smalltable}[3][h]{
    \begin{table}[#1]                     % The file must contain one element: \begin{table}...\end{table}.
        \begin{center}\begin{small}
            \input{tables/#2}
			\caption{#3\label{#2}}
        \end{small}\end{center}
    \end{table}
    }

% New command to add long tables. Usage:
% \newlongtable[H]{filename-tabular}{caption}  
\newcommand{\newlongtable}[3][!ht]{
    \begin{center}                        % The file must contain one element: \begin{longtable}...\end{longtable}.
        \input{tables/#2}
		\captionof{table}{#3\label{#2}}
    \end{center}
    \addtocounter{table}{-1}  %Decrease the table counter by one to get the right caption.
}
